 123def record_and_plot_multiome_graph_stats(
 124    data, relationships, dataset_name="Dataset", bin_size=5, save_path="./"
 125):
 126    """
 127    Records statistics for a heterogeneous graph dataset and plots specified metrics for multiple relationships,
 128    following the logic of counting the occurrences of target nodes in edge indices.
 129
 130    Parameters:
 131    - data: The data object or subgraph, typically in PyTorch Geometric format.
 132      Expected to be a bidirectional graph.
 133    - relationships: List of tuples representing the relationships to process.
 134      Each tuple is (source_node_type, edge_type, target_node_type).
 135    - dataset_name: A name identifier for the dataset (e.g., 'Train', 'Validation', or 'Full Dataset').
 136    - bin_size: Size of the bins for neighbor count distributions in the histograms.
 137    - save_path: Directory path to save the plots.
 138    """
 139
 140    # Initialize a dictionary to store stats
 141    stats = {}
 142
 143    if dataset_name is not None:
 144        print(f"{dataset_name} Statistics:")
 145
 146    for src_type, rel_type, dst_type in relationships:
 147        # Print relationship info
 148        print(f"\nProcessing relationship: ({src_type}, {rel_type}, {dst_type})")
 149
 150        # Check if the edge type exists in the data
 151        if (src_type, rel_type, dst_type) not in data.edge_types:
 152            print(
 153                f"Edge type ({src_type}, {rel_type}, {dst_type}) not found in data. Skipping."
 154            )
 155            continue
 156
 157        # Get the edge index for the relationship
 158        edge_index = data[src_type, rel_type, dst_type].edge_index
 159        # src_indices = edge_index[0].tolist()
 160        dst_indices = edge_index[1].tolist()
 161
 162        # Count the occurrences of target nodes (following your logic)
 163        target_neighbor_count = Counter(dst_indices)
 164        target_neighbor_values = list(target_neighbor_count.values())
 165        max_target_neighbors = (
 166            max(target_neighbor_values) if target_neighbor_values else 0
 167        )
 168        bins = np.arange(0, max_target_neighbors + bin_size, bin_size)
 169
 170        # Plot histogram for target neighbor counts
 171        plt.figure(figsize=(10, 6))
 172        plt.hist(target_neighbor_values, bins=bins, edgecolor="black")
 173        plt.title(
 174            f"Distribution of {src_type} connections per {dst_type} - {dataset_name}"
 175        )
 176        plt.xlabel(f"Number of {src_type} connections per {dst_type}")
 177        plt.ylabel(f"Number of {dst_type} nodes")
 178        plt.savefig(
 179            f"{save_path}/Distribution_of_{src_type}_connections_per_{dst_type}_{dataset_name}_{rel_type}.png"
 180        )
 181        plt.close()
 182
 183        # Similarly, process the reverse edge if it exists
 184        reverse_edge_type = (dst_type, f"rev_{rel_type}", src_type)
 185        if reverse_edge_type in data.edge_types:
 186            print(f"Processing reverse relationship: {reverse_edge_type}")
 187            rev_edge_index = data[reverse_edge_type].edge_index
 188            rev_dst_indices = rev_edge_index[1].tolist()
 189
 190            # Count the occurrences of target nodes in the reverse edge
 191            rev_target_neighbor_count = Counter(rev_dst_indices)
 192            rev_target_neighbor_values = list(rev_target_neighbor_count.values())
 193            max_rev_target_neighbors = (
 194                max(rev_target_neighbor_values) if rev_target_neighbor_values else 0
 195            )
 196            bins = np.arange(0, max_rev_target_neighbors + bin_size, bin_size)
 197
 198            # Plot histogram for reverse target neighbor counts
 199            plt.figure(figsize=(10, 6))
 200            plt.hist(rev_target_neighbor_values, bins=bins, edgecolor="black")
 201            plt.title(
 202                f"Distribution of {dst_type} connections per {src_type} - {dataset_name}"
 203            )
 204            plt.xlabel(f"Number of {dst_type} connections per {src_type}")
 205            plt.ylabel(f"Number of {src_type} nodes")
 206            plt.savefig(
 207                f"{save_path}/Distribution_of_{dst_type}_connections_per_{src_type}_{dataset_name}_{rel_type}.png"
 208            )
 209            plt.close()
 210        else:
 211            print(f"Reverse edge type {reverse_edge_type} not found in data.")
 212
 213        # Record stats
 214        stats_key = f"{src_type}_{rel_type}_{dst_type}"
 215        stats[stats_key] = {
 216            "num_edges": edge_index.size(1),
 217            "target_neighbor_counts": target_neighbor_values,
 218        }
 219
 220        if reverse_edge_type in data.edge_types:
 221            rev_stats_key = f"{dst_type}_rev_{rel_type}_{src_type}"
 222            stats[rev_stats_key] = {
 223                "num_edges": rev_edge_index.size(1),
 224                "target_neighbor_counts": rev_target_neighbor_values,
 225            }
 226
 227    return stats

 231    """
 232    Summarizes the overall statistics of an epoch based on a list of batch DataFrames
 233    and plots the distribution of gene and cell neighbor counts.

 235    Parameters:
 236    - df_list: List of DataFrames where each DataFrame contains statistics for a batch,
 237      with keys ['num_cells', 'num_genes', 'num_links', 'gene_neighbor_count', 'cell_neighbor_count'].

 239    Returns:
 240    - epoch_stats: A dictionary containing aggregated statistics for the epoch.

 242
 243    # Initialize accumulators for stats

 245    total_genes = 0
 246    total_links = 0
 247    all_gene_neighbor_counts = []
 248    all_cell_neighbor_counts = []

 254        total_genes += batch_stats["num_genes"]

 360def empirical_coverage(
 361    true_values,
 362    pred_means,
 363    pred_vars,
 364    fig_name="cell_express_gene_empirical_coverage",
 365    save_path="./",
 366):
 367    # Calculate empirical coverage
 368    confidence_levels = np.linspace(0.5, 0.99, 10)  # Coverage levels (50% to 99%)
 369    empirical_coverage = []
 370
 371    for level in confidence_levels:
 372        z = norm.ppf((1 + level) / 2)  # Z-value for two-tailed interval
 373        lower_bound = pred_means - z * np.sqrt(pred_vars)
 374        upper_bound = pred_means + z * np.sqrt(pred_vars)
 375
 376        # Check how many true values fall within the interval
 377        within_interval = np.sum(
 378            (true_values >= lower_bound) & (true_values <= upper_bound)
 379        )
 380        empirical_coverage.append(within_interval / len(true_values))
 381
 382    # Plot empirical coverage
 383    plt.figure(figsize=(8, 6))
 384    plt.plot(
 385        confidence_levels, empirical_coverage, label="Empirical Coverage", marker="o"
 386    )
 387    plt.plot(
 388        confidence_levels,
 389        confidence_levels,
 390        label="Ideal Calibration",
 391        linestyle="--",
 392        color="red",
 393    )
 394    plt.xlabel("Nominal Coverage")
 395    plt.ylabel("Empirical Coverage")
 396    plt.title("Calibration Plot (Empirical Coverage)")
 397    plt.legend()
 398    plt.grid(True)
 399    plt.savefig(f"{save_path}/{fig_name}.png", dpi=300)
 400    plt.close()
